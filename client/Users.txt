POST - /users/singup
	tozihat: baraie sabtenam kardan. niaz ba auth nadare. faghat niaz be body dare.
	file json ba attribute haye (name, lastname, number, email, password) ferestade mishe.
	response: etelate user + token

POST - /users/login
	tozihat: baraie login kardane. niaz be auth nadare. faghat niaz be body dare.
	file json ba attribute haye (email, password) ferestade mishe.
	har bar user login mikone ye token sakhte mishe.
	response: etelate user + token

POST - /users/logout
	tozihat: baraie logout kardan. NIAZ BE AUTH DARE. be body niaz nadare.
	vaghti yek user logout mikone , tokeni ke baraie AUTH esteafde mikone az database az arraye token oon user hazf mishe.
	response: response khasi felan nadare

POST - /users/logoutALL
	tozihat: baraie logout kardan az hame device haye motasel shode. NIAZ BE AUTH DARE. be body niaz nadare.
	haman kare logout mikone faghat ba in tafavot ke tamami token haye marboot be array oon usero az database pak mikone.
	(ghablie faghat tokeni ke baraie auth bood pak mikard vali in ba har tokeni age zade beshe , kolle token haro pak mikone.)
	response: response khasi felan nadare

GET - /users/me
	tozihat: baraie didane profile user hast. NIAZ BE AUTH DARE. be body niaz nadare.
	user mitone ba tokenesh etelate marboor be profilesho bebine.
	response: response khasi felan nadare

PATCH - /users/me
	tozihat: baraie update kardan profile user hast. NIAZ BE AUTH DARE. NIAZ BE BODY DARE.
	file json ba ba harkodam az attribute haye zamane sabtenam.
	masalan age user mikhad (name, lastname) esho avaz kone baiad in 2 attribute estafde kone
	{
		"name": "...",
		"lastname": "..."
	}

	va ya age mikhad faghat password avaz kone
	{
		"password": "...",
	}

	ALL ATTRIBUTES : 'name', 'lastname', 'number', 'email', 'password', 'gender', 'bday', 'codinglanguage', 'education'

	beharhal harchio bekhad avaz kone oon too mizare xD . 
	response: etelate user (dar vaghe etelate user bad az update shodanesh)
	
DELETE - /users/me
	tozihat: baraie pak kardan user az database. NIAZ BE AUTH DARE. be body niaz nadare.
	tamami etelate user az database pak mishe va dg fard nemitone login kone, va mitavan dobare ba hamoon etalat signup jadid anjam dad.


POST - /users/me/avatar
	tozihat: baraie add kardane avatar be profile user. NIAZ BE AUTH DARE. NIAZ BE BODY DARE.
	file body baiad "form-data" bashe va hatman - key - on baiad "avatar" bashe , value ham baiad oon akse bashe.
	MESAL :       key="avatar"     value="example.jpg"
	response: dar soorate dorost ferestade shodan fagaht status code 200 miad va hich massage dgei nemiad vali age 
			  ertebat fail she response ye file JSON maid ke error marboot too oonja zekr shode.

DELETE - /users/me/avatar
	tozihat: baraie pak kardane avatar dar profile user. NIAZ BE AUTH DARE. be body niaz nadare.
	kare khasi niaz nadare , fagaht ba oon token ye request dade mishe va aks pak mishe
	response: hich response khassi nadare va faghat dar soorat bargharr shodan ertebat status code 200 mide

GET - /users/:id/avatar
	tozihat: baraie didane avatar profile, NIAZ BE HICH CHIZ NADARE.
	hame mitoonam az tarighe url bebinanesh.
	dar kol chize khassi nadare.(baghie tozihatesh too skype dadam)
